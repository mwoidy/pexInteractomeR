Fig 3B: Fraction of pex proteins expressed in tissue (expressed + TiP)

p <- protein_expr %>% 
  filter(!is.na(expression)) %>% 
  group_by(tissue, expression) %>% 
  summarise(n = n()) %>% 
  mutate(fraction = n/sum(n)) %>% 
  mutate(n = if_else(expression == "expressed", sum(n), n)) %>% 
  mutate(fraction = if_else(expression == "expressed", n/172, fraction)) %>%
  mutate(sorting = if_else(expression == "expressed", n, 0)) %>%
  arrange(tissue, desc(expression)) %>%
  ungroup() %>%
  complete(tissue, expression, fill=list(n=0, fraction=0, sorting=0)) %>%
  ggplot(aes(x=reorder(tissue, -sorting), y = fraction, fill=expression)) +
  geom_bar(stat = "identity", position = position_identity()) +
  #geom_text(

Function that takes fun_genes and a gene_list and outputs n of function per gene of gene_list

getFunction <- function(fun_genes, gene_list) {
  fun_counts <- fun_genes %>% filter(symbol %in% gene_list) %>% count(geneFunction)
  return(fun_counts)
}
